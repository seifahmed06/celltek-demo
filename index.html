export default async function handler(req, res) { if (req.method !== 'POST') { return res.status(405).json({ error: 'Method not allowed' }); } const { text, voiceId } = req.body; if (!text) { return res.status(400).json({ error: 'Missing text' }); } const ELEVENLABS_API_KEY = process.env.ELEVENLABS_API_KEY; if (!ELEVENLABS_API_KEY) { console.error('Missing ELEVENLABS_API_KEY'); return res.status(500).json({ error: 'Server configuration error' }); } // Default to Rachel if no voiceId provided const finalVoiceId = voiceId || '21m00Tcm4TlvDq8ikWAM'; const url = `https://api.elevenlabs.io/v1/text-to-speech/${finalVoiceId}`; try { const response = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json', 'xi-api-key': ELEVENLABS_API_KEY, }, body: JSON.stringify({ text, model_id: 'eleven_monolingual_v1', voice_settings: { stability: 0.5, similarity_boost: 0.75 }, }), }); if (!response.ok) { const errorText = await response.text(); console.error('ElevenLabs error:', errorText); return res.status(response.status).json({ error: 'TTS service error' }); } const audioBuffer = await response.arrayBuffer(); res.setHeader('Content-Type', 'audio/mpeg'); res.send(Buffer.from(audioBuffer)); } catch (err) { console.error('Proxy error:', err); res.status(500).json({ error: 'Internal server error' }); } }
